auto lo
iface lo inet loopback

# -----------------------------------------------------------
# NODO: PVE (Node 1) - IP: 10.10.10.11
# PATH: /etc/network/interfaces
# -----------------------------------------------------------

iface enp1s0f0 inet manual
# Porta 1 (VLAN 10 - Server)

iface enp1s0f1 inet manual
# Porta 2 (VLAN 20 - Client)

# -----------------------------------------------------------
# PORTA DI SERVIZIO (DIRECT ACCESS - VLAN 1 Admin Subnet)
# -----------------------------------------------------------
auto nic0
iface eno3 inet static
    address 192.168.100.11/24
    # NESSUN GATEWAY. Collega PC con IP 192.168.100.x

# -----------------------------------------------------------
# BRIDGE MANAGEMENT & SERVER (VLAN 10)
# -----------------------------------------------------------
auto vmbr0
iface vmbr0 inet static
    address 10.10.10.11/24
    gateway 10.10.10.1
    bridge-ports enp1s0f0
    bridge-stp off
    bridge-fd 0

# -----------------------------------------------------------
# BRIDGE CLIENT (VLAN 20)
# -----------------------------------------------------------
auto vmbr20
iface vmbr20 inet manual
    bridge-ports enp1s0f1
    bridge-stp off
    bridge-fd 0
